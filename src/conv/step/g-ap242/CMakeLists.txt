# local includes
include_directories(
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}/../g-step
  )

  ${express_srcs}
  ../BRLCADWrapper.cpp
  ../g-step/AP203.cpp

  librt

  libbn

  stepcore

  set_target_properties(g-ap242 PROPERTIES COMPILE_DEFINITIONS "${gap242_IMPORTS}")
  set_property(SOURCE
