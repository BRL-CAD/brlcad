  ${BRLCAD_BINARY_DIR}/include
  ${BRLCAD_SOURCE_DIR}/include
  ${DM_INCLUDE_DIRS}
  ${NMG_INCLUDE_DIRS}
  ${FB_INCLUDE_DIRS}
  ${GED_INCLUDE_DIRS}
  ${BN_INCLUDE_DIRS}
  ${BU_INCLUDE_DIRS}
  ${TCL_INCLUDE_DIRS}
  ${TK_INCLUDE_DIRS}
  )

  ${PNG_INCLUDE_DIRS}

  libfuncs.c

  tclcad_init.c

  )

endif (TARGET itk_pkgIndex)
